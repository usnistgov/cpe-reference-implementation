27  package gov.nist.secauto.cpe.matching;
28  
29  import static org.junit.jupiter.api.Assertions.assertFalse;
30  import static org.junit.jupiter.api.Assertions.assertTrue;
31  
32  import gov.nist.secauto.cpe.common.LogicalValue;
33  import gov.nist.secauto.cpe.common.WellFormedName;
34  import gov.nist.secauto.cpe.matching.CPENameMatcher;
35  import gov.nist.secauto.cpe.naming.CPENameUnbinder;
36  
37  import org.junit.jupiter.api.Test;
38  
39  import java.text.ParseException;
40  
41  class CPENameMatcherTest {
42  
43    @Test
44    void test() throws ParseException {
45      // Examples.
46      WellFormedName wfn = new WellFormedName("a", "microsoft", "internet_explorer", "8\\.0\\.6001", "beta",
47          LogicalValue.ANY, "sp2", null, null, null, null);
48      WellFormedName wfn2 = new WellFormedName("a", "microsoft", "internet_explorer", LogicalValue.ANY, LogicalValue.ANY,
49          LogicalValue.ANY, LogicalValue.ANY, LogicalValue.ANY, LogicalValue.ANY, LogicalValue.ANY, LogicalValue.ANY);
50  
51      assertFalse(CPENameMatcher.isDisjoint(wfn, wfn2));
52      assertFalse(CPENameMatcher.isEqual(wfn, wfn2));
53      assertTrue(CPENameMatcher.isSubset(wfn, wfn2)); // wfn2 is a subset of wfn
54      assertFalse(CPENameMatcher.isSuperset(wfn, wfn2));
55  
56      wfn = CPENameUnbinder.unbindFS("cpe:2.3:a:adobe:*:9.*:*:PalmOS:*:*:*:*:*");
57      wfn2 = CPENameUnbinder.unbindURI("cpe:/a::Reader:9.3.2:-:-");
58      assertTrue(CPENameMatcher.isDisjoint(wfn, wfn2)); // true, wfn2 and wfn are disjoint
59      assertFalse(CPENameMatcher.isEqual(wfn, wfn2));
60      assertFalse(CPENameMatcher.isSubset(wfn, wfn2));
61      assertFalse(CPENameMatcher.isSuperset(wfn, wfn2));
62    }
