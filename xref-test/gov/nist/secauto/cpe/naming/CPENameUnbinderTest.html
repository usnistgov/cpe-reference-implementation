27  package gov.nist.secauto.cpe.naming;
28  
29  import static org.junit.jupiter.api.Assertions.assertEquals;
30  
31  import gov.nist.secauto.cpe.common.LogicalValue;
32  import gov.nist.secauto.cpe.common.WellFormedName;
33  import gov.nist.secauto.cpe.common.WellFormedName.Attribute;
34  
35  import org.junit.jupiter.api.Test;
36  
37  import java.text.ParseException;
38  
39  class CPENameUnbinderTest {
40  
41    @Test
42    void test() throws ParseException {
43      // A few examples.
44      WellFormedName wfn = CPENameUnbinder.unbindURI("cpe:/a:microsoft:internet_explorer%01%01%01%01:?:beta");
45      assertEquals("a", wfn.get(Attribute.PART));
46      assertEquals("microsoft", wfn.get(Attribute.VENDOR));
47      assertEquals("internet_explorer????", wfn.get(Attribute.PRODUCT));
48      assertEquals("?", wfn.get(Attribute.VERSION));
49      assertEquals("beta", wfn.get(Attribute.UPDATE));
50      assertEquals(LogicalValue.ANY, wfn.get(Attribute.EDITION));
51      assertEquals(LogicalValue.ANY, wfn.get(Attribute.LANGUAGE));
52      assertEquals(LogicalValue.ANY, wfn.get(Attribute.SW_EDITION));
53      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_SW));
54      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_HW));
55      assertEquals(LogicalValue.ANY, wfn.get(Attribute.OTHER));
56  
57      wfn = CPENameUnbinder.unbindURI("cpe:/a:microsoft:internet_explorer:8.%2a:sp%3f");
58      assertEquals("a", wfn.get(Attribute.PART));
59      assertEquals("microsoft", wfn.get(Attribute.VENDOR));
60      assertEquals("internet_explorer", wfn.get(Attribute.PRODUCT));
61      assertEquals("8\\.\\*", wfn.get(Attribute.VERSION));
62      assertEquals("sp\\?", wfn.get(Attribute.UPDATE));
63      assertEquals(LogicalValue.ANY, wfn.get(Attribute.EDITION));
64      assertEquals(LogicalValue.ANY, wfn.get(Attribute.LANGUAGE));
65      assertEquals(LogicalValue.ANY, wfn.get(Attribute.SW_EDITION));
66      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_SW));
67      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_HW));
68      assertEquals(LogicalValue.ANY, wfn.get(Attribute.OTHER));
69  
70      wfn = CPENameUnbinder.unbindURI("cpe:/a:microsoft:internet_explorer:8.%02:sp%01");
71      assertEquals("a", wfn.get(Attribute.PART));
72      assertEquals("microsoft", wfn.get(Attribute.VENDOR));
73      assertEquals("internet_explorer", wfn.get(Attribute.PRODUCT));
74      assertEquals("8\\.*", wfn.get(Attribute.VERSION));
75      assertEquals("sp?", wfn.get(Attribute.UPDATE));
76      assertEquals(LogicalValue.ANY, wfn.get(Attribute.EDITION));
77      assertEquals(LogicalValue.ANY, wfn.get(Attribute.LANGUAGE));
78      assertEquals(LogicalValue.ANY, wfn.get(Attribute.SW_EDITION));
79      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_SW));
80      assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_HW));
81      assertEquals(LogicalValue.ANY, wfn.get(Attribute.OTHER));
82  
83      wfn = CPENameUnbinder.unbindURI("cpe:/a:hp:insight_diagnostics:7.4.0.1570::~~online~win2003~x64~");
84      assertEquals("a", wfn.get(Attribute.PART));
85      assertEquals("hp", wfn.get(Attribute.VENDOR));
86      assertEquals("insight_diagnostics", wfn.get(Attribute.PRODUCT));
87      assertEquals("7\\.4\\.0\\.1570", wfn.get(Attribute.VERSION));
88      assertEquals(LogicalValue.ANY, wfn.get(Attribute.UPDATE));
89      assertEquals(LogicalValue.ANY, wfn.get(Attribute.EDITION));
90      assertEquals(LogicalValue.ANY, wfn.get(Attribute.LANGUAGE));
91      assertEquals("online", wfn.get(Attribute.SW_EDITION));
92      assertEquals("win2003", wfn.get(Attribute.TARGET_SW));
93      assertEquals("x64", wfn.get(Attribute.TARGET_HW));
94      assertEquals(LogicalValue.ANY, wfn.get(Attribute.OTHER));
95  
96      wfn = CPENameUnbinder.unbindFS("cpe:2.3:a:micr\\?osoft:internet_explorer:8.0.6001:beta:*:*:*:*:*:*");
97      assertEquals("a", wfn.get(Attribute.PART));
98      assertEquals("micr\\?osoft", wfn.get(Attribute.VENDOR));
99      assertEquals("internet_explorer", wfn.get(Attribute.PRODUCT));
100     assertEquals("8\\.0\\.6001", wfn.get(Attribute.VERSION));
101     assertEquals("beta", wfn.get(Attribute.UPDATE));
102     assertEquals(LogicalValue.ANY, wfn.get(Attribute.EDITION));
103     assertEquals(LogicalValue.ANY, wfn.get(Attribute.LANGUAGE));
104     assertEquals(LogicalValue.ANY, wfn.get(Attribute.SW_EDITION));
105     assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_SW));
106     assertEquals(LogicalValue.ANY, wfn.get(Attribute.TARGET_HW));
107     assertEquals(LogicalValue.ANY, wfn.get(Attribute.OTHER));
108   }
