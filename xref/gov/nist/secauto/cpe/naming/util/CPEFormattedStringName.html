52  package gov.nist.secauto.cpe.naming.util;
53  
54  import gov.nist.secauto.cpe.common.WellFormedName;
55  import gov.nist.secauto.cpe.naming.CPENameUnbinder;
56  
57  import java.text.ParseException;
58  
59  /**
60   * A concrete implementation of a formatted string unbound to a {@link WellFormedName}.
61   */
62  public class CPEFormattedStringName extends AbstractCPEName {
63  
64    /**
65     * Constructs a formatted string bound CPE name based on the provided formatted string.
66     * 
67     * @param formattedString
68     *          the formatted string
69     * @throws ParseException
70     *           if the formatted string is not valid
71     */
72    public CPEFormattedStringName(String formattedString) throws ParseException {
73      super(CPENameUnbinder.unbindFS(formattedString));
74    }
75  
76    /**
77     * Gets the underlying {@link WellFormedName} as a formatted string.
78     *
79     * @return the {@link WellFormedName} as a CPE formatted string
80     * 
81     * @see #toFormattedString()
82     */
83    @Override
84    public String toString() {
85      return toFormattedString();
86    }
87  
88  }
