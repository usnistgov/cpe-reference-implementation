74    // Define the attributes that correspond to the seven components in a v2.2. CPE.
75    public static final WellFormedName.Attribute[] URI_ATTRIBUTES
76        = { WellFormedName.Attribute.PART, WellFormedName.Attribute.VENDOR, WellFormedName.Attribute.PRODUCT,
77            WellFormedName.Attribute.VERSION, WellFormedName.Attribute.UPDATE, WellFormedName.Attribute.EDITION,
78            // requires packing
79            WellFormedName.Attribute.LANGUAGE };

81    /**
82     * Binds a {@link WellFormedName} object to a URI.
83     * 
84     * @param wfn
85     *          WellFormedName to be bound to URI
86     * @return URI binding of WFN
87     */
88    public static String bindToURI(WellFormedName wfn) {
89  
90      // Initialize the output with the CPE v2.2 URI prefix.
91      String uri = "cpe:/";
92  
93      // Iterate over the well formed name
94      for (WellFormedName.Attribute attr : URI_ATTRIBUTES) {
95        String value = "";
96        if (WellFormedName.Attribute.EDITION.equals(attr)) {
97          // Call the pack() helper function to compute the proper
98          // binding for the edition element.
99          String edition = bindValueForURI(wfn.get(WellFormedName.Attribute.EDITION));
100         String swEdition = bindValueForURI(wfn.get(WellFormedName.Attribute.SW_EDITION));
101         String targetSoftware = bindValueForURI(wfn.get(WellFormedName.Attribute.TARGET_SW));
102         String targetHardware = bindValueForURI(wfn.get(WellFormedName.Attribute.TARGET_HW));
103         String other = bindValueForURI(wfn.get(WellFormedName.Attribute.OTHER));
104         value = pack(edition, swEdition, targetSoftware, targetHardware, other);
105       } else {
106         // Get the value for attr in wfn, then bind to a string
107         // for inclusion in the URI.
108         value = bindValueForURI(wfn.get(attr));
109       }
110       // Append value to the URI then add a colon.
111       uri = Utilities.strcat(uri, value, ":");
112     }
113     // Return the URI string, with trailing colons trimmed.
114     return trim(uri);
115   }

117   /**
118    * Top-level function used to bind WFN w to formatted string.
119    * 
120    * @param wfn
121    *          WellFormedName to bind
122    * @return Formatted String
123    */
124   public static String bindToFS(WellFormedName wfn) {
125     // Initialize the output with the CPE v2.3 string prefix.
126     String fs = "cpe:2.3:";
127     for (WellFormedName.Attribute attr : WellFormedName.Attribute.values()) {
128       String value = bindValueForFS(wfn.get(attr));
129       fs = Utilities.strcat(fs, value);
130       // add a colon except at the very end
131       if (!WellFormedName.Attribute.OTHER.equals(attr)) {
132         fs = Utilities.strcat(fs, ":");
133       }
134     }
135     return fs;
136   }

138   /**
139    * Convert the value v to its proper string representation for insertion to formatted string.
140    * 
141    * @param value
142    *          value to convert
143    * @return Formatted value
144    */
145   private static String bindValueForFS(Object value) {
146     if (value instanceof LogicalValue) {
147       LogicalValue../gov/nist/secauto/cpe/common/LogicalValue.html#LogicalValue">LogicalValue logicalValue = (LogicalValue) value;
148       // The value NA binds to a blank.
149       if (LogicalValue.ANY.equals(logicalValue)) {
150         return "*";
151       }
152       // The value NA binds to a single hyphen.
153       if (LogicalValue.NA.equals(logicalValue)) {
154         return "-";
155       }
156     }
157     return processQuotedChars((String) value);
158   }

160   /**
161    * Inspect each character in the provided string, and escape as required.
162    * <p>
163    * Certain non-alpha characters pass thru without escaping into the result, but most retain
164    * escaping.
165    * 
166    * @param str
167    *          the string to process
168    * @return the processed string result
169    */
170   private static String processQuotedChars(String str) {
171     String result = "";
172     int index = 0;
173     while (index < Utilities.strlen(str)) {
174       String ch = Utilities.substr(str, index, index + 1);
175       if (!ch.equals("\\")) {
176         // unquoted characters pass thru unharmed.
177         result = Utilities.strcat(result, ch);
178       } else {
179         // escaped characters are examined.
180         String nextchr = Utilities.substr(str, index + 1, index + 2);
181         // the period, hyphen and underscore pass unharmed.
182         if (nextchr.equals(".") || nextchr.equals("-") || nextchr.equals("_")) {
183           result = Utilities.strcat(result, nextchr);
184           index = index + 2;
185           continue;
186         } else {
187           // all others retain escaping.
188           result = Utilities.strcat(result, "\\", nextchr);
189           index = index + 2;
190           continue;
191         }
192       }
193       index = index + 1;
194     }
195     return result;
196   }

198   /**
199    * Converts a value to the proper string for including in a CPE v2.2-conformant URI. The logical
200    * value ANY binds to the blank in the 2.2-conformant URI.
201    * 
202    * @param value
203    *          the value to be converted
204    * @return the converted string
205    */
206   private static String bindValueForURI(Object value) {
207     if (value instanceof LogicalValue) {
208       LogicalValue../gov/nist/secauto/cpe/common/LogicalValue.html#LogicalValue">LogicalValue logicalValue = (LogicalValue) value;
209       // The value NA binds to a blank.
210       if (LogicalValue.ANY.equals(logicalValue)) {
211         return "";
212       }
213       // The value NA binds to a single hyphen.
214       if (LogicalValue.NA.equals(logicalValue)) {
215         return "-";
216       }
217     }
218 
219     // If we get here, we're dealing with a string value.
220     return transformForURI((String) value);
221   }

223   /**
224    * Scans an input string and performs a series of transformations to convert the string to a bound
225    * URI form.
226    * <p>
227    * The following transformations are performed:
228    * <ul>
229    * <li>Pass alphanumeric characters thru untouched</li>
230    * <li>Percent-encode quoted non-alphanumerics as needed</li>
231    * <li>Unquoted special characters are mapped to their special forms</li>
232    * </ul>
233    * 
234    * @param str
235    *          string to be transformed
236    * @return transformed string
237    */
238   private static String transformForURI(String str) {
239     String result = "";
240     int idx = 0;
241 
242     while (idx < Utilities.strlen(str)) {
243       // Get the idx'th character of s.
244       String thischar = Utilities.substr(str, idx, idx + 1);
245       // Alphanumerics (incl. underscore) pass untouched.
246       if (Utilities.isAlphanum(thischar)) {
247         result = Utilities.strcat(result, thischar);
248         idx = idx + 1;
249         continue;
250       }
251       // Check for escape character.
252       if (thischar.equals("\\")) {
253         idx = idx + 1;
254         String nxtchar = Utilities.substr(str, idx, idx + 1);
255         result = Utilities.strcat(result, pctEncode(nxtchar));
256         idx = idx + 1;
257         continue;
258       }
259       // Bind the unquoted '?' special character to "%01".
260       if (thischar.equals("?")) {
261         result = Utilities.strcat(result, "%01");
262       }
263       // Bind the unquoted '*' special character to "%02".
264       if (thischar.equals("*")) {
265         result = Utilities.strcat(result, "%02");
266       }
267       idx = idx + 1;
268     }
269     return result;
270   }

272   /**
273    * Returns the appropriate percent-encoding of character c. Certain characters are returned without
274    * encoding.
275    * 
276    * @param ch
277    *          the single character string to be encoded
278    * @return the percent encoded string
279    */
280   private static String pctEncode(String ch) {
281     if (ch.equals("!")) {
282       return "%21";
283     }
284     if (ch.equals("\"")) {
285       return "%22";
286     }
287     if (ch.equals("#")) {
288       return "%23";
289     }
290     if (ch.equals("$")) {
291       return "%24";
292     }
293     if (ch.equals("%")) {
294       return "%25";
295     }
296     if (ch.equals("&")) {
297       return "%26";
298     }
299     if (ch.equals("'")) {
300       return "%27";
301     }
302     if (ch.equals("(")) {
303       return "%28";
304     }
305     if (ch.equals(")")) {
306       return "%29";
307     }
308     if (ch.equals("*")) {
309       return "%2a";
310     }
311     if (ch.equals("+")) {
312       return "%2b";
313     }
314     if (ch.equals(",")) {
315       return "%2c";
316     }
317     // bound without encoding.
318     if (ch.equals("-")) {
319       return ch;
320     }
321     // bound without encoding.
322     if (ch.equals(".")) {
323       return ch;
324     }
325     if (ch.equals("/")) {
326       return "%2f";
327     }
328     if (ch.equals(":")) {
329       return "%3a";
330     }
331     if (ch.equals(";")) {
332       return "%3b";
333     }
334     if (ch.equals("<")) {
335       return "%3c";
336     }
337     if (ch.equals("=")) {
338       return "%3d";
339     }
340     if (ch.equals(">")) {
341       return "%3e";
342     }
343     if (ch.equals("?")) {
344       return "%3f";
345     }
346     if (ch.equals("@")) {
347       return "%40";
348     }
349     if (ch.equals("[")) {
350       return "%5b";
351     }
352     if (ch.equals("\\")) {
353       return "%5c";
354     }
355     if (ch.equals("]")) {
356       return "%5d";
357     }
358     if (ch.equals("^")) {
359       return "%5e";
360     }
361     if (ch.equals("`")) {
362       return "%60";
363     }
364     if (ch.equals("{")) {
365       return "%7b";
366     }
367     if (ch.equals("|")) {
368       return "%7c";
369     }
370     if (ch.equals("}")) {
371       return "%7d";
372     }
373     if (ch.equals("~")) {
374       return "%7d";
375     }
376     // Shouldn't reach here, return original character
377     return ch;
378   }

380   /**
381    * Packs the values of the five arguments into the single edition component. If all the values are
382    * blank, the function returns a blank.
383    * 
384    * @param edition
385    *          edition string
386    * @param swEdition
387    *          software edition string
388    * @param targetSoftware
389    *          target software string
390    * @param targetHardware
391    *          target hardware string
392    * @param other
393    *          other edition information string
394    * @return the packed string, or blank
395    */
396   private static String pack(String edition, String swEdition, String targetSoftware, String targetHardware,
397       String other) {
398     if (swEdition.equals("") && targetSoftware.equals("") && targetHardware.equals("") && other.equals("")) {
399       // All the extended attributes are blank, so don't do
400       // any packing, just return ed.
401       return edition;
402     }
403     // Otherwise, pack the five values into a single string
404     // prefixed and internally delimited with the tilde.
405     return Utilities.strcat("~", edition, "~", swEdition, "~", targetSoftware, "~", targetHardware, "~", other);
406   }

408   /**
409    * Removes trailing colons from the URI.
410    * 
411    * @param str
412    *          the string to be trimmed
413    * @return the trimmed string
414    */
415   private static String trim(String str) {
416     String s1 = Utilities.reverse(str);
417     int idx = 0;
418     for (int i = 0; i != Utilities.strlen(s1); i++) {
419       if (Utilities.substr(s1, i, i + 1).equals(":")) {
420         idx = idx + 1;
421       } else {
422         break;
423       }
424     }
425     // Return the substring after all trailing colons,
426     // reversed back to its original character order.
427     return Utilities.reverse(Utilities.substr(s1, idx, Utilities.strlen(s1)));
428   }
